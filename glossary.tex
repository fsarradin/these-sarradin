\chapter*{Notation}

\begin{tabular}{p{.33\linewidth}l}
$\mathbbm{R}$ \dotfill      & Ensemble des réels \\
$\mathbbm{Z}$ \dotfill      & Ensemble des entiers relatifs \\
$\mathbbm{L}_{\sigma}$ \dotfill
                            & Maillage carrée de résolution $\sigma$ \\
$L_{\sigma}(S)$ \dotfill    & Projection de $S$ sur $\mathbbm{L}_{\sigma}$ \\
$B^d(r,p)$ \dotfill         & Boule ouverte de centre $p$ et de rayon
                              $r$ pour la distance $d$ \\
$B^E(r,p)$ \dotfill         & Boule ouverte euclidienne de centre $p$
                              et de rayon $r$ \\
$\mathcal{A}(S)$ \dotfill   & Aire de l'objet $S$ \\
$\mathcal{P}(S)$ \dotfill   & Périmètre de l'objet $S$ \\
$\partial S$ \dotfill       & Contour de l'objet $S$ \\
$\mathrm{int}(S)$ \dotfill  & Intérieur de l'objet $S$ \\
$\overline{S}$ \dotfill     & Fermeture de l'objet $S$ \\
$S^C$ \dotfill              & Complémentaire de l'objet $S$ \\
$||\cdot||$ ou $L_2(\cdot)$ \dotfill
                            & Norme euclidienne \\
$d_E(\cdot,\cdot)$ \dotfill & Distance euclidienne \\
$d_4(\cdot,\cdot)$ \dotfill & Distance de Manhattan \\
$d_8(\cdot,\cdot)$ \dotfill & Distance échiquier \\
$d_{\langle\cdots\rangle}(\cdot,\cdot)$ \dotfill
                            & Distance de chanfrein pour le masque
                              $\langle\cdots\rangle$ \\
$d_h(\cdot,\cdot)$ \dotfill & Distance hyperbolique \\
%$d_F(\cdot,\cdot)$ \dotfill & Distance de Fréchet \\
$d_{\mathcal{H}}(\cdot,\cdot)$ \dotfill
                            & Distance de Hausdorff \\
$\mathcal{H}(\cdot|\cdot)$ \dotfill
                            & Distance de Hausdorff relative \\
%$d_{\mathcal{H}_h}(\cdot,\cdot)$ \dotfill
%                            & Distance de Hausdorff hyperbolique \\
%$\mathcal{H}_h(\cdot|\cdot)$ \dotfill
%                            & Distance de Hausdorff hyperbolique relative \\
$\langle\cdot,\cdot\rangle$ \dotfill
                            & Produit scalaire \\
$|\cdot|$ \dotfill          & Nombre cardinal \\
\end{tabular}
